222 v_pint get_frontier(const v_real& W, const v_pint& X) {
223  pint n{ X.size() };
224  real sumY{ 0.0 };
225  real sumY_vec{ W[X[n-1]] - W[X[0]] };
226  pint sumX_vec{ X[n-1] - X[0] };
227 
228  for (pint i = 0; i < n; i++) {
229  sumY += W[X[i]];
230  }
231  real scaling{ (sumX_vec / 2.0) / sumY };
232  real sumk{ 0.0 };
233  real x;
234  real y;
235 
236  v_real Y(n);
237  Y[0] = W[X[0]] * scaling;
238  for (pint i = 1; i < n; i++) {
239  Y[i] = W[X[i]] * scaling;
240  x = X[i] - X[i - 1];
241  y = Y[i] - Y[i - 1];
242  sumk += y / (x * sqrt(x * x + y * y));
243  }
244  real r{ 1.0 / (sumk / (n - 1)) };
245  real rr{ r * r };
246  pint idx1{ 0 };
247  pint idx2{ 1 };
248  v_pint frontierX = { X[0] };
249  point pc;
250  bool empty;
251 
252  while (idx2 < n) {
253  pc = get_circle(X[idx1], Y[idx1], X[idx2], Y[idx2], r);
254  empty = true;
255  for (pint i = idx2 + 1; i < n; i++) {
256  if (pow(pc.x - X[i], 2.0) + pow(pc.y - Y[i], 2.0) < rr) {
257  empty = false;

260  }
261  }
