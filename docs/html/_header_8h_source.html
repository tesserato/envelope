149  real q{ sqrt(pow(x1 - x0, 2.0) + pow(y1 - y0, 2.0)) };
150  real c{ sqrt(r * r - pow(q / 2.0, 2.0)) };
151

153  real y3{ (y0 + y1) / 2.0 };
